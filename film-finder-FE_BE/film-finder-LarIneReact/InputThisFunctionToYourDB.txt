CREATE OR REPLACE FUNCTION update_film_rating()
RETURNS TRIGGER AS $$
BEGIN
    -- Update the film's rating based on the average of all reviews for the film
    UPDATE films
    SET rating_film = (
        SELECT ROUND(AVG(rating_user), 1)
        FROM reviews
        WHERE film_id = NEW.film_id
    )
    WHERE film_id = NEW.film_id;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;
